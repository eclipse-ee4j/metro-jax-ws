213     protected List<String> getCommonArgs() throws MojoExecutionException {
214         List<String> commonArgs = new ArrayList<String>();
215 
216         if (!isDefaultSrc(getSourceDestDir()) || keep) {
217             commonArgs.add("-keep");
218             commonArgs.add("-s");
219             commonArgs.add("'" + getSourceDestDir().getAbsolutePath() + "'");
220             if (!getSourceDestDir().mkdirs() && !getSourceDestDir().exists()) {
221                 getLog().warn("Cannot create directory: " + getSourceDestDir().getAbsolutePath());
222             }
223             addSourceRoot(getSourceDestDir().getAbsolutePath());
224         }
225 
226         File destDir = getDestDir();
227         if (!destDir.mkdirs() && !destDir.exists()) {
228             getLog().warn("Cannot create directory: " + destDir.getAbsolutePath());
229         }
230         commonArgs.add("-d");
231         commonArgs.add("'" + destDir.getAbsolutePath() + "'");
232 
233         if (verbose) {
234             commonArgs.add("-verbose");
235         }
236 
237         if (isArgSupported("-encoding")) {
238             if (encoding != null) {
239                 maybeUnsupportedOption("-encoding", encoding, commonArgs);
240             } else {
241                 getLog().warn("Using platform encoding (" + System.getProperty("file.encoding") + "), build is platform dependent!");
242             }
243         }
244 
245         if (isExtensionOn()) {
246             commonArgs.add("-extension");
247         }
248 
249         if(getXnocompile()){
250             commonArgs.add("-Xnocompile");
251         }
252 
253         // add additional command line options
254         if (args != null) {
255             for (String arg : args) {
256                 commonArgs.add(arg);
257             }
258         }
259         return commonArgs;
260     }

315     protected void exec(List<String> args) throws MojoExecutionException {
316         StreamConsumer sc = new DefaultConsumer();
317         try {
318             Commandline cmd = new Commandline();
319             if (executable != null) {
320                 if (executable.isFile() && executable.canExecute()) {
321                     cmd.setExecutable(executable.getAbsolutePath());
322                     if (getExtraClasspath() !=  null) {
323                         cmd.createArg().setLine("-cp");
324                         cmd.createArg().setValue(getExtraClasspath());
325                     }
326                 } else {
327                     throw new MojoExecutionException("Cannot execute: " + executable.getAbsolutePath());
328                 }
329             } else {
330                 cmd.setExecutable(new File(new File(System.getProperty("java.home"), "bin"), getJavaExec()).getAbsolutePath());
331                 // add additional JVM options
332                 if (vmArgs != null) {
333                     for (String arg : vmArgs) {
334                         cmd.createArg().setLine(arg);
335                     }
336                 }
337                 String[] classpath = getCP();
338                 cmd.createArg().setValue("-cp");
339                 cmd.createArg().setValue(classpath[1]);
340                 cmd.createArg().setLine("org.jvnet.jax_ws_commons.jaxws.Invoker");
341                 cmd.createArg().setLine(getMain());
342                 String extraCp = getExtraClasspath();
343                 String cp = extraCp != null ? extraCp + File.pathSeparator : "";
344                 cp += classpath[0];
345                 try {
346                     File pathFile = createPathFile(cp);
347                     cmd.createArg().setLine("-pathfile " + pathFile.getAbsolutePath());
348                 } catch (IOException ioe) {
349                     //creation of temporary file can fail, in such case just put everything on cp
350                     cmd.createArg().setValue("-cp");
351                     cmd.createArg().setValue(cp);
352                 }
353             }
354             cmd.setWorkingDirectory(project.getBasedir());
355             for (String arg : args) {
356                 cmd.createArg().setLine(arg);
357             }
358             String fullCommand = cmd.toString();
359             if (isWindows() && 8191 <= fullCommand.length()) {
360                 getLog().warn("Length of the command is limitted to 8191 characters but it has "
361                         + fullCommand.length() + " characters.");
362                 getLog().warn(fullCommand);
363             } else {
364                 getLog().debug(fullCommand);
365             }
366             if (CommandLineUtils.executeCommandLine(cmd, sc, sc) != 0) {
367                 throw new MojoExecutionException("Mojo failed - check output");
368             }
369         } catch (DependencyResolutionException | CommandLineException dre) {
370             throw new MojoExecutionException(dre.getMessage(), dre);
371         }
372     }

394     private String[] getCP() throws DependencyResolutionException {
395         Map<String, org.eclipse.aether.artifact.Artifact> cp = new HashMap<String, org.eclipse.aether.artifact.Artifact>();
396         Plugin p = pluginDescriptor.getPlugin();
397         boolean toolsFound = false;
398         for (Dependency d : p.getDependencies()) {
399             DependencyResult result = DependencyResolver.resolve(d,
400                     new ExclusionFilter(d.getExclusions()),
401                     pluginRepos, repoSystem, repoSession);
402             sortArtifacts(result, cp);
403             if (containsTools(cp.keySet())) {
404                 toolsFound = true;
405             }
406         }
407         for (String dep : getExtraDependencies()) {
408             DependencyResult result = DependencyResolver.resolve(
409                     pluginDescriptor.getArtifactMap().get(dep),
410                     toolsFound ? new DepFilter(getExtraArtifactIDs()) : null,
411                     pluginRepos, repoSystem, repoSession);
412             sortArtifacts(result, cp);
413         }
414         if (!containsTools(cp.keySet())) {
415             DependencyResult result = DependencyResolver.resolve(
416                     pluginDescriptor.getArtifactMap().get("com.sun.xml.ws:jaxws-tools"),
417                     null, pluginRepos, repoSystem, repoSession);
418             sortArtifacts(result, cp);
419         }
420         StringBuilder sb = getCPasString(cp.values());
421         //add custom invoker
422         String invokerPath = AbstractJaxwsMojo.class.getProtectionDomain().getCodeSource().getLocation().toExternalForm();
423         try {
424             invokerPath = new URI(invokerPath).getPath();
425             sb.append(invokerPath);
426         } catch (URISyntaxException ex) {
427             throw new RuntimeException(ex);
428         }
429         sb.append(File.pathSeparator);
430         //don't forget tools.jar
431         File toolsJar = new File(System.getProperty("java.home"), "../lib/tools.jar");
432         if (!toolsJar.exists()) {
433             toolsJar = new File(System.getProperty("java.home"), "lib/tools.jar");
434         }
435         if (toolsJar.exists()) {
436             sb.append(toolsJar.getAbsolutePath());
437             sb.append(File.pathSeparator);
438         }
439         getLog().debug("getCP sb: " + sb);
440         getLog().debug("getCP iv: " + invokerPath);
441         return new String[]{sb.substring(0, sb.length() - 1), invokerPath};
442     }

448     private File createPathFile(String cp) throws IOException {
449         File f = File.createTempFile("jax-ws-mvn-plugin-cp", ".txt");
450         if (f.exists() && f.isFile()) {
451             if (!f.delete()) {
452                 //this should not happen
453                 getLog().warn("cannot remove obsolete classpath setting file: " + f.getAbsolutePath());
454             }
455         }
456         Properties p = new Properties();
457         p.put("cp", cp.replace(File.separatorChar, '/'));
458         getLog().debug("stored classpath: " + cp.replace(File.separatorChar, '/'));
459         FileOutputStream fos = null;
460         try {
461             fos = new FileOutputStream(f);
462             p.store(fos, null);
463         } catch (IOException ex) {
464             logger.log(Level.SEVERE, null, ex);
465         } finally {
466             if (fos != null) {
467                 try {
468                     fos.close();
469                 } catch (IOException ex) {
470                     logger.log(Level.SEVERE, null, ex);
471                 }
472             }
473         }
474         return f;
475     }

481     private StringBuilder getCPasString(Collection<org.eclipse.aether.artifact.Artifact> artifacts) {
482         StringBuilder sb = new StringBuilder();
483         for (org.eclipse.aether.artifact.Artifact a : artifacts) {
484             sb.append(a.getFile().getAbsolutePath());
485             sb.append(File.pathSeparator);
486         }
487         return sb;
488     }

490     private void sortArtifacts(DependencyResult result, Map<String, org.eclipse.aether.artifact.Artifact> cp) {
491         PreorderNodeListGenerator nlg = new PreorderNodeListGenerator();
492         result.getRoot().accept(nlg);
493         for (org.eclipse.aether.artifact.Artifact a : nlg.getArtifacts(false)) {
494             cp.put(a.getGroupId() + ":" + a.getArtifactId(), a);
495         }

505     private static class DepFilter implements DependencyFilter {
506 
507         private final Set<Dep> toExclude = new HashSet<Dep>();
508 
509         public DepFilter(String[] artifacts) {
510             if (artifacts != null) {
511                 for (String a : artifacts) {
512                     int i = a.indexOf(':');
513                     toExclude.add(new Dep(a.substring(0, i), a.substring(i + 1)));
514                 }
515             }
516         }
517 
518         @Override
519         public boolean accept(DependencyNode node, List<DependencyNode> parents) {
520             org.eclipse.aether.artifact.Artifact a = node.getDependency().getArtifact();
521             return !toExclude.contains(new Dep(a.getGroupId(), a.getArtifactId()));
522         }

524         private static class Dep {
525 
526             private final String groupId;
527             private final String artifactId;
528 
529             public Dep(String groupId, String artifactId) {
530                 this.groupId = groupId;
531                 this.artifactId = artifactId;
532             }
533 
534             @Override
535             public int hashCode() {
536                 int hash = 5;
537                 hash = 37 * hash + (this.groupId != null ? this.groupId.hashCode() : 0);
538                 return hash;
539             }
540 
541             @Override
542             public boolean equals(Object obj) {
543                 if (obj == null) {
544                     return false;
545                 }
546                 if (getClass() != obj.getClass()) {
547                     return false;
548                 }
549                 final Dep other = (Dep) obj;
550                 if ((this.groupId == null)
551                         ? (other.groupId != null)
552                         : !this.groupId.equals(other.groupId)) {
553                     return false;
554                 }
555                 //startsWith here is intentional
556                 return !((this.artifactId == null)
557                         ? (other.artifactId != null)
558                         : !this.artifactId.startsWith(other.artifactId));
559             }
560         }
