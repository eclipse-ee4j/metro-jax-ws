18  package org.jvnet.jax_ws_commons.jaxws;
19  
20  import java.io.File;
21  import org.apache.maven.plugins.annotations.LifecyclePhase;
22  import org.apache.maven.plugins.annotations.Mojo;
23  import org.apache.maven.plugins.annotations.Parameter;
24  import org.apache.maven.plugins.annotations.ResolutionScope;
25  
26  /**
27   * Parses wsdl and binding files and generates Java code needed to access it.
28   *
29   * @author Kohsuke Kawaguchi
30   */
31  @Mojo(name = "wsimport", defaultPhase = LifecyclePhase.GENERATE_SOURCES, requiresDependencyResolution = ResolutionScope.RUNTIME)
32  public class MainWsImportMojo extends WsImportMojo {
33  
34      /**
35       * Specify where to place output generated classes. Use <code>xnocompile</code>
36       * to turn this off.
37       */
38      @Parameter(defaultValue = "${project.build.outputDirectory}")
39      private File destDir;
40  
41      /**
42       * Specify where to place generated source files, keep is turned on with this option.
43       */
44      @Parameter(defaultValue = "${project.build.directory}/generated-sources/wsimport")
45      private File sourceDestDir;
46  
47      /**
48       * Specify where to generate JWS implementation file.
49       */
50      @Parameter(defaultValue = "${project.build.sourceDirectory}")
51      private File implDestDir;
52  
53      /**
54       * Either ${build.outputDirectory} or ${build.testOutputDirectory}.
55       */
56      @Override
57      protected File getDestDir() {
58          return destDir;
59      }
60  
61      @Override
62      protected File getSourceDestDir() {
63          return sourceDestDir;
64      }
65  
66      @Override
67      protected File getDefaultSrcOut() {
68          return new File(project.getBuild().getDirectory(), "generated-sources/wsimport");
69      }
70  
71      @Override
72      protected File getImplDestDir() {
73          return implDestDir;
74      }
75  }
